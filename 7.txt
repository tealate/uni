using UnityEngine;

class PlaneEquation {
    public float a, b, c, d;

    public PlaneEquation(Vector3 point1, Vector3 point2, Vector3 point3) {
        // 2つのベクトルを作成
        Vector3 v1 = point2 - point1;
        Vector3 v2 = point3 - point1;

        // 平面の法線ベクトルを計算
        Vector3 normal = Vector3.Cross(v1, v2).normalized;

        // ax + by + cz + d = 0の形式に変換
        a = normal.x;
        b = normal.y;
        c = normal.z;
        d = -Vector3.Dot(normal, point1);
    }
}
